      name: Satoshi_MemeRush
      bnb_wallet: '0xABCD...1234'
      token_targets:
        - MemeRush
        - BNB
      trade_limits:
        buy_max_usd: 100
        sell_trigger_pct: 7.5
      strategy:
        type: 'momentum'
        params:
          lookback_minutes: 13
          trend_threshold: 2.1

        openai_enabled: true
        claude_enabled: true
        volatility_threshold: 'high'
      notifications:
